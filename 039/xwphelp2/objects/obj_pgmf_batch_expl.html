Esempio 1: 
ECHO OFF
REM Questo script crea una copia di riserva del file CONFIG.SYS 
REM nella cartella C:\OS2.
ECHO Copia del file CONFIG.SYS in C:\OS2.
ECHO Se un file CONFIG.SYS modificato non dovesse funzionare, ricopiare questo su C:.
COPY C:\CONFIG.SYS C:\OS2\CONFIG.SYS
ECHO Script terminato.
Esempio 2: 
ECHO OFF
REM Questo script stampa un file di testo standard.
REM Per usarlo, digitare il nome del file da stampare dopo
REM aver digitato il nome dello script.
REM Se lo script si chiama STAMPA.CMD ed il file da stampare
REM LEGGIMI, digitare STAMPA LEGGIMI
ECHO Stampa in corso di %1.
TYPE %1 > lPT1
ECHO ON
